No hay recomendaciones en esta categoría.

"; return; } filtradas.forEach(({titulo, categoria, descripcion, url, imagen}) => { const div = document.createElement('div'); div.classList.add('recomendacion'); div.innerHTML = `

${titulo}

${categoria}

${descripcion}

Ver producto `; contenedor.appendChild(div); }); } // Inicial render renderizarRecomendaciones(); // Agregar recomendación form.addEventListener('submit', e => { e.preventDefault(); const titulo = document.getElementById('titulo').value.trim(); const categoria = document.getElementById('categoria').value; const descripcion = document.getElementById('descripcion').value.trim(); const url = document.getElementById('url').value.trim(); let imagen = document.getElementById('imagen').value.trim(); // Si no se pone imagen, dejamos vacío para el placeholder if(imagen && !imagen.startsWith('http')) { alert('Por favor ingresa una URL válida para la imagen.'); return; } if (!imagen) { imagen = ''; } recomendaciones.push({titulo, categoria, descripcion, url, imagen}); localStorage.setItem('recomendaciones', JSON.stringify(recomendaciones)); renderizarRecomendaciones(getCategoriaActiva()); form.reset(); }); // Filtrado por categorías categoriaBtns.forEach(btn => { btn.addEventListener('click', () => { categoriaBtns.forEach(b => b.classList.remove('active')); btn.classList.add('active'); renderizarRecomendaciones(btn.getAttribute('data-categoria')); }); }); function getCategoriaActiva() { const activo = document.querySelector('.categoria-lista button.active'); return activo ? activo.getAttribute('data-categoria') : "Todas"; }
